int main(void) {
    // Configura PD6 e PD7 como saída
    DDRD |= (1 << PD6) | (1 << PD7);

    // Liga os dois pinos (coloca em nível alto)
    PORTD |= (1 << PD6) | (1 << PD7);

    while (1) {
        // Apaga apenas o PD6
        PORTD &= ~(1 << PD6);

        _delay_ms(500);

        // Religa PD6 para repetir o ciclo
        PORTD |= (1 << PD6);

        _delay_ms(500);
    }

    return 0;
}
