int main(void) {
    // Configurando PD5 como entrada
    DDRD &= ~(1 << PD5);

    // Ativando pull-up no PD5
    PORTD |= (1 << PD5);

    while (1) {
        // Verifica o estado do pino PD5
        if (PIND & (1 << PD5)) {
            // Se PD5 está em nível alto
            // (nada sera pressionado se for botão com pull-up)
        } else {
            // Se PD5 está em nível baixo
            // (botão pressionado, indo para o GND)
        }
    }

    return 0;
}
